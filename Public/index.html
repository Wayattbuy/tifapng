⚙️ Configuration Supabase requise

Pour utiliser tifapng, configurez votre projet Supabase :

1. Créez un projet sur supabase.com

2. Remplacez les valeurs dans le code :

const SUPABASE_URL = 'https://VOTRE-PROJET.supabase.co';
const SUPABASE_ANON_KEY = 'VOTRE-ANON-KEY';

3. Exécutez ce SQL dans Supabase :

-- Profils utilisateurs
CREATE TABLE profiles (
  id UUID REFERENCES auth.users PRIMARY KEY,
  username TEXT UNIQUE,
  bio TEXT,
  avatar_url TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Publications
CREATE TABLE posts (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES auth.users NOT NULL,
  title TEXT NOT NULL,
  description TEXT,
  image_url TEXT,
  category TEXT DEFAULT 'mignon',
  privacy TEXT DEFAULT 'public',
  except_users TEXT[],
  likes_count INT DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Likes
CREATE TABLE likes (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  post_id UUID REFERENCES posts NOT NULL,
  user_id UUID REFERENCES auth.users NOT NULL,
  UNIQUE(post_id, user_id)
);

-- Commentaires
CREATE TABLE comments (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  post_id UUID REFERENCES posts NOT NULL,
  user_id UUID REFERENCES auth.users NOT NULL,
  content TEXT NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Comptes enfants (KIDS)
CREATE TABLE children (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  parent_id UUID REFERENCES auth.users NOT NULL,
  name TEXT NOT NULL,
  age INT,
  avatar TEXT DEFAULT '🐱',
  time_limit_minutes INT DEFAULT 60,
  allowed_categories TEXT[],
  allowed_channels UUID[],
  is_active BOOLEAN DEFAULT true
);

-- RLS Policies (activer RLS sur chaque table)
ALTER TABLE profiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE posts ENABLE ROW LEVEL SECURITY;
ALTER TABLE likes ENABLE ROW LEVEL SECURITY;
ALTER TABLE comments ENABLE ROW LEVEL SECURITY;
ALTER TABLE children ENABLE ROW LEVEL SECURITY;

-- Policies simplifiées
CREATE POLICY "Public profiles" ON profiles FOR SELECT USING (true);
CREATE POLICY "Own profile" ON profiles FOR ALL USING (auth.uid() = id);
CREATE POLICY "Public posts" ON posts FOR SELECT USING (privacy = 'public' OR privacy = 'kids' OR user_id = auth.uid());
CREATE POLICY "Own posts" ON posts FOR INSERT WITH CHECK (auth.uid() = user_id);
CREATE POLICY "Own posts update" ON posts FOR UPDATE USING (auth.uid() = user_id);
CREATE POLICY "Own posts delete" ON posts FOR DELETE USING (auth.uid() = user_id);
CREATE POLICY "Likes read" ON likes FOR SELECT USING (true);
CREATE POLICY "Likes write" ON likes FOR ALL USING (auth.uid() = user_id);
CREATE POLICY "Comments read" ON comments FOR SELECT USING (true);
CREATE POLICY "Comments write" ON comments FOR INSERT WITH CHECK (auth.uid() = user_id);
CREATE POLICY "Children own" ON children FOR ALL USING (auth.uid() = parent_id);

4. Activez le Storage dans Supabase et créez un bucket cat-photos en Public.
